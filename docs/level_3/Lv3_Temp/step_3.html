온도 설정하기

온도를 측정하고 유지하는 것까지 알았습니다. 이제는 온도를 직접 설정하는 방법에 대해서 알아보겠습니다.
원하는 온도를 세세하게 설정하기엔 복잡함이 있음으로, 미리 재료에 적절한 온도 값을 저장하고, 그 값을 불러내어 목표온도로 설정하는 방식을 구현 해보는 것이 목표입니다.

재료에 맞는 온도 값을 미리 배열로 저장하고 불러오기 위해, 변수와 배열을 선언합니다.
1int tempValueA0 = 0; // A0 신호 값 저장용
2int curTemp        = 0; // 계산된 온도
3int targetTemp  = 0; // 설정 온도
4int presetIndex = 0; // 저장된 설정 온도의 현재 순번
5
6                  // 기본, PCL, PLA
7int presetTemp[3] = { 0,  60,  200 };

배열과 함께 변수가 2개 추가되었습니다. 먼저 현재온도가 아닌 목표온도(설정온도)가 무슨 값을 가지고 있는지 저장하고, 알기 위해 targetTemp 변수를 만들었습니다. 이어서 배열은 길이가 3개 입니다. 각각 0도 60도 200도를 설정할 수 있도록 3개의 항목이 배열에 저장되어 있습니다.
이 배열 현재의 위치를 가르키고 있는 숫자를 인덱스(index)라고 합니다. 인덱스의 뜻은 지표, 색인, 지수등을 뜻합니다. 프로그래밍에서는 배열[인덱스]와 같은 형태로 사용됩니다. 인덱스 변수와 배열은 코드로는 다음과 같습니다.
1targetTemp = presetTemp[presetIndex];

presetTemp의 배열의 presetIndex 의 위치의 값을 설정온도(targetTemp)로 저장하겠다는 표시입니다.
여기에서 버튼을 누를 때마다 presetIndex가 1씩 증가(혹은 감소)하면, targetTemp 의 값도 계속해서 변경되어질 것입니다.

버튼을 누를 때마다 설정온도가 변경되도록 전체 코드를 작성해봅니다. 그리고 아래 코드와 비교해 봅니다.
  1int tempValueA0 = 0; // A0 신호 값 저장용
  2int curTemp = 0;
  3int targetTemp = 0;
  4int presetIndex = 0;
  5bool isHeating = false; // 예열 상태를 확인하는 확인하는 bool 변수
  6
  7                 // 기본, PCL, PLA
  8int presetTemp[3] = { 0,  60,  200 };
  9
 10int temptable[23][2] =
 11{
 12    {1023,0},
 13    {1008,10},
 14    {994,20},
 15    {990,30},
 16    {985,40},
 17    {983,50},
 18    {981,60},
 19    {978,70},
 20    {975,80},
 21    {965,90},
 22    {959,100},
 23    {952,110},
 24    {948,120},
 25    {941,130},
 26    {932,140},
 27    {920,150},
 28    {908,160},
 29    {890,170},
 30    {875,180},
 31    {845,190},
 32    {820,200},
 33    {790,210},
 34    {765,220}
 35}; // 온도테이블
 36
 37void setup() {
 38    // put your setup code here, to run once:
 39    Serial.begin(9600);
 40
 41    pinMode(11,INPUT_PULLUP);
 42    pinMode(12,INPUT_PULLUP);
 43
 44    pinMode(9,OUTPUT);
 45}
 46
 47// 신호 값을 보정하여 온도 값을 추측해내는 계산 함수
 48int tempCali(int valueA0)
 49{
 50    float ratioTemp;
 51
 52    for(int i = 0; i<23; i++)
 53    {
 54        if(temptable[i][0] <= valueA0)
 55        {
 56            ratioTemp = ((float)valueA0 - temptable[i][0])/(temptable[i-1][0] - temptable[i][0]);
 57
 58            return temptable[i][1] - ratioTemp*(temptable[i][1] - temptable[i-1][1]);
 59        }
 60
 61    }
 62}
 63
 64// 아날로그 A0 핀의 신호 값을 디지털 9번핀(열선)이 HIGH인 상태에서 체크하도록 하는 함수
 65void checkA0()
 66{
 67    digitalWrite(9, HIGH); // 예열 시작
 68    delay(1);
 69    tempValueA0 = analogRead(A0); // 아날로그 신호 값을 tempValueA0 저장
 70    if(!isHeating)
 71    {
 72        digitalWrite(9, LOW); // 예열 종료
 73    }
 74}
 75
 76void loop() {
 77    // put your main code here, to run repeatedly:
 78    checkA0(); // 아날로그 A0의 신호 값을 가져오는 함수
 79
 80    curTemp = tempCali(tempValueA0); // 온도 보상 함수 호출
 81
 82    Serial.print("신호 값 : ");
 83    Serial.print(tempValueA0);
 84    Serial.print(", 현재 온도 값 : ");
 85    Serial.print(curTemp);
 86    Serial.print(", 설정 온도 값 : ");
 87    Serial.println(targetTemp);
 88
 89    if(curTemp > targetTemp)
 90    {
 91        digitalWrite(9, LOW); // 예열 종료
 92        delay(5); // 약간의 대기시간 추가
 93
 94        isHeating = false;
 95    }
 96    else
 97    {
 98        digitalWrite(9, HIGH); // 예열 시작
 99        delay(5); // 약간의 대기시간 추가
100
101        isHeating = true;
102    }
103
104    if(digitalRead(11)==LOW)
105    {
106        presetIndex++;
107        targetTemp = presetTemp[presetIndex];
108
109        if(presetIndex>2)
110        {
111          presetIndex = 0;
112        }
113        delay(100);
114    }
115    else if(digitalRead(12)==LOW)
116    {
117        presetIndex--;
118        targetTemp = presetTemp[presetIndex];
119
120        if(presetIndex<0)
121        {
122          presetIndex = 2;
123        }
124        delay(100);
125    }
126
127}
